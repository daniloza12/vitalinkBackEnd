│                                                                              │

│ Contact      { fullName, phone, personalEmail, workEmail }                   │

│ workEmail, city, district }                                                  │
│ ContactVisibility   { fullName, phone, personalEmail, workEmail }            │

│ Services (interfaces + impl)                                                 │
│                                                                              │
│ AuthService / AuthServiceImpl                                                │
│                                                                              │
│ register(email, password) → Account:                                         │
│   1. Verificar que email no exista                                           │
│   2. Hashear password con SHA-256                                            │
│   3. Generar securityAccount = SHA-256(email +                               │
│ System.currentTimeMillis()).substring(0,32)                                  │
│   4. Generar UUID para id                                                    │
│   5. Setear role=USER, createdAt=now()                                       │
│   6. Guardar y retornar Account                                              │
│                                                                              │
│ login(email, password) → Account:                                            │
│   1. Buscar por email                                                        │
│   2. Hashear password recibida                                               │
│   3. Comparar hash almacenado                                                │
│   4. Si coincide, retornar Account; sino throw 401                           │
│                                                                              │
│ AccountService / AccountServiceImpl                                          │
│                                                                              │
│ getAll()                              → List<Account>                        │
│ getById(id)                           → Account (throw 404 si no existe)     │
│ getBySecurityAccount(secAccount)      → Account (throw 404 si no existe)     │
│ update(id, data)                      → Account (actualizar campos no nulos) │
│ delete(id)                            → void                                 │
│                                                                              │
│ ProfileService / ProfileServiceImpl                                          │
│                                                                              │
│ getByAccountId(accountId)             → Profile (throw 404 si no existe)     │
│ create(profile)                       → Profile (throw 409 si ya existe)     │
│ update(accountId, profile)            → Profile (throw 404 si no existe)     │
│                                                                              │
│ ---                                                                          │
│ Controllers                                                                  │
│                                                                              │
│ AuthController — POST /api/v1/auth                                           │
│                                                                              │
│ @PostMapping("/register")  → 201 Created + Account body                      │
│ @PostMapping("/login")     → 200 OK + Account body (401 si credenciales      │
│ inválidas)                                                                   │
│                                                                              │
│ AccountController — /api/v1/accounts                                         │
│                                                                              │
│ @GetMapping            → 200 + List<Account>                                 │
│ @GetMapping("/{id}")   → 200 + Account                                       │
│ @GetMapping("/security/{securityAccount}") → 200 + Account                   │
│ @PutMapping("/{id}")   → 200 + Account actualizado                           │
│ @DeleteMapping("/{id}")→ 204 No Content                                      │
│                                                                              │
│ ProfileController — /api/v1/profiles                                         │
│                                                                              │
│ @GetMapping("/{accountId}")  → 200 + Profile                                 │
│ @PostMapping                 → 201 + Profile creado                          │
│ @PutMapping("/{accountId}")  → 200 + Profile actualizado                     │
